% Preamble contents, to be included in main.tex

% Quick way to always get a name
\makeatletter
\def\@authorname{\ifx\@empty\fromsig%
		\fromname%
    \else \fromsig \fi}
\makeatother

% Don't wrap name
\makeatletter
\settowidth{\indentedwidth}{\@authorname}
\makeatother

% Custom signature space after the closing remark
% (Default: no space)
\newcommand{\sigspace}{\medskipamount}
\makeatletter
\renewcommand{\closing}[1]{\par\nobreak\vspace{\parskip}%
	\stopbreaks
	\noindent
	\ifx\@empty\fromaddress\else
		\hspace*{\longindentation}\fi
	\parbox{2\indentedwidth}{\raggedright
		\ignorespaces #1\\[\sigspace]%
		\@authorname\strut}%
	\par}
\makeatother
